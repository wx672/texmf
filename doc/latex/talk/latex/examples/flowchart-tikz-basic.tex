\documentclass[tikz]{standalone}

\usetikzlibrary{calc,shapes,arrows,positioning,graphs}

\begin{document}

% Define block styles
\tikzstyle{check} = [diamond, aspect=2, inner sep=0pt, align=center]%
\tikzstyle{block} = [rectangle, align=center]%

\begin{tikzpicture}[auto, %
  every node/.style={draw},%
  every path/.style={draw, -latex'}]
  
  % Place nodes
  \node [block, rounded corners] (start) {Start};%
  \node [block, below=of start] (one) {1. Request\\feature};%
  \node [block, below=of one] (two) {2. Define\\feature};%
  \node [check, below=of two] (chk) {Feature\\approved?};%
  \node [block, below=of chk] (three) {3. Design\\feature};%
  \node [block, below=of three, rounded corners] (done) {Done};%
  \node [block, right=of three] (threeb) {3b. Withdraw\\feature};%

  \node [ellipse callout, callout absolute pointer={(two.east)},%
  draw, above right=3mm of two, align=center, scale=.5] {This step includes\\an approval process};
  
  % Draw edges
  \path (start) edge (one)
  (one) edge (two)
  (two) edge (chk)
  (chk) edge node[draw=none] {yes} (three)
  (three) edge (done);%
  \path (chk) -| node [near start, draw=none] {no} (threeb);%
  \path (threeb) |- ($(three.south)+(0,-5mm)$);
\end{tikzpicture}
\end{document}
